Number of checks required to check if n is prime:  (sqrt(n)/2)+1

==> Print Number of Primes.
************************************************
#include<stdio.h>
#include<stdbool.h>
#include<math.h>
bool isPrime(long long n){
    if(n<=1) return false;
    if((n&1)==0 && n!=2) return false;
    long long t=sqrt(n);
    for(long long i=3;i<=t;i+=2){
        if(n%i==0) return false;
    }
    return true;
}//T(n) => 7+4(sqrt(n)/2)
int main(){
    long long m,n,c=0;
    scanf("%lld %lld",&m,&n);
    if(m<=2 && n>=2) {c++; m=3;}
    else if((m&1)==0) {m++;}
    for(long long i=m;i<=n;i+=2){
        if(isPrime(i)) c++;
    }
    printf("%lld",c);
}
