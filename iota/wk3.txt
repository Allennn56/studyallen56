const int irPin=10;
void setup() {

  // put your setup code here, to run once:
  pinMode(irPin,INPUT);
  Serial.begin(9600);

}

void loop() {
  // put your main code here, to run repeatedly:
  int val;
  val=digitalRead(irPin);
  Serial.println(val);
  if(val==0)
  Serial.println("Obstacle is detected ");
  else
  Serial.println("no obstacle");
  delay(1000);

}
